      try {

    checkApiKeys() {
      const candidates = [
        { name: 'ANTHROPIC_API_KEY', check: () => !!process.env.ANTHROPIC_API_KEY },
        { name: 'OPENAI_API_KEY',    check: () => !!process.env.OPENAI_API_KEY },
        { name: 'BRAVE_API_KEY',     check: () => !!process.env.BRAVE_API_KEY },

            catch (e) { return false; }
          }
        }
      ];
      const present = candidates.filter(c => c.check()).map(c => c.name);
      if (present.length === 0) {
        this.warnings.push('No AI provider API key found (set ANTHROPIC_API_KEY or OPENAI_API_KEY)');
        return false;
      }

    }

      const pkgPath = path.join(this.cwd, 'package.json');

      if (!fs.existsSync(nmPath)) {
        this.warnings.push('node_modules not found — run npm install');
        return false;
      }
      return true;
    }
