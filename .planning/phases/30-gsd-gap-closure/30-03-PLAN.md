function readCostConfig(cwd) {
  const configPath = require('path').join(cwd, '.planning', 'config.json');
  if (!require('fs').existsSync(configPath)) return defaultCostConfig();
  try {
    const raw = JSON.parse(require('fs').readFileSync(configPath, 'utf8'));
    return Object.assign(defaultCostConfig(), raw.cost_tracking || {});
  } catch (e) { return defaultCostConfig(); }
}
function defaultCostConfig() {

    - opts: { ceiling?: Number, warning_threshold?: Number } — defaults from getConfig() if not provided
    - Gets aggregate().total.cost
    - If ceiling is null or not a number, returns { status: 'ok', message: 'No budget set' }
    - If total >= ceiling: returns { status: 'exceeded', message: `Budget ceiling $${ceiling} exceeded ($${total.toFixed(4)} spent)`, total, ceiling }
    - If total >= ceiling * (warning_threshold / 100): returns { status: 'warning', message: `${percentUsed}% of budget used`, total, ceiling, percentUsed }
    - Otherwise: returns { status: 'ok', message: 'Within budget', total, ceiling, percentUsed }
    - Does NOT call process.exit() or print to console — caller decides what to do with the result

    - Sets config.cost_tracking.budget = ceiling
    - If warning_threshold provided, sets config.cost_tracking.warning_threshold = warning_threshold
    - Writes config.json back
    - Returns void (synchronous)

    aggregate() reduce — exact implementation:
    ```js
    aggregate(filter = {}) {
      const zero = () => ({ total: { cost: 0, tokens: 0 }, by_phase: {}, by_provider: {} });
      if (!fs.existsSync(this.metricsPath)) return zero();
      let data;
      try { data = JSON.parse(fs.readFileSync(this.metricsPath, 'utf8')); }
      catch (e) { return zero(); }
      let entries = (data.entries || []);
      if (filter.phase !== undefined) entries = entries.filter(e => e.phase == filter.phase);
      if (filter.milestone) entries = entries.filter(e => e.milestone === filter.milestone);
      if (filter.provider) entries = entries.filter(e => e.provider === filter.provider);
      return entries.reduce((acc, e) => {
        const phaseKey = String(e.phase || 'unknown');
        if (!acc.by_phase[phaseKey]) acc.by_phase[phaseKey] = { cost: 0, tokens: 0 };
        acc.by_phase[phaseKey].cost   += e.cost_usd || 0;
        acc.by_phase[phaseKey].tokens += (e.input_tokens || 0) + (e.output_tokens || 0);
        const provKey = e.provider || 'unknown';
        if (!acc.by_provider[provKey]) acc.by_provider[provKey] = { cost: 0 };
        acc.by_provider[provKey].cost += e.cost_usd || 0;
        acc.total.cost   += e.cost_usd || 0;
        acc.total.tokens += (e.input_tokens || 0) + (e.output_tokens || 0);
        return acc;
      }, zero());
    }
    ```
